        // Function to calculate recitation method
        function calculateRecitationMethod(totalAbjad) {
            if (totalAbjad === 0) return "کوئی عدد درج کریں";
            
            // 1. Darood Pak count (divide by 9)
            let daroodCount = totalAbjad % 9;
            if (daroodCount === 0) daroodCount = 9;
            
            // 2. Time of recitation (divide by 5)
            const timeRemainder = totalAbjad % 5;
            let timeOfDay = "";
            switch(timeRemainder) {
                case 1: timeOfDay = "فجر کے بعد"; break;
                case 2: timeOfDay = "ظہر کے بعد"; break;
                case 3: timeOfDay = "عصر کے بعد"; break;
                case 4: timeOfDay = "مغرب کے بعد"; break;
                case 0: timeOfDay = "عشاء کے بعد"; break;
                default: timeOfDay = "عشاء کے بعد";
            }
            
            // 3. Start day (divide by 7)
            const dayRemainder = totalAbjad % 7;
            let startDay = "";
            switch(dayRemainder) {
                case 1: startDay = "سوموار (پیر)"; break;
                case 2: startDay = "منگل"; break;
                case 3: startDay = "بدھ"; break;
                case 4: startDay = "جمعرات"; break;
                case 5: startDay = "جمعہ"; break;
                case 6: startDay = "ہفتہ"; break;
                case 0: startDay = "اتوار"; break;
                default: startDay = "اتوار";
            }
            
            // 4. Calculate digit sum for optional recitation count
            const digitSum = totalAbjad.toString().split('').reduce((sum, digit) => sum + parseInt(digit), 0);
            
            return `
                <p><strong>📿 درود پاک:</strong> ${daroodCount} مرتبہ</p>
                <p><strong>⏰ وقت:</strong> ${timeOfDay}</p>
                <p><strong>📖 تعداد (مکمل):</strong> ${totalAbjad} مرتبہ روزانہ</p>
                <p><strong>🌙 شروع کا دن:</strong> ${startDay}</p>
                <p><strong>✨ تعداد (اختیاری):</strong> ${digitSum} مرتبہ روزانہ (کم از کم)</p>
                <p style="font-size: 0.8rem; margin-top: 8px; color: var(--gold);">*بہترین طریقہ: ${totalAbjad} مرتبہ روزانہ پڑھیں</p>
            `;
        }

        function calculateIsm(val) {
            const container = document.getElementById('virtues-container');
            const ismDisplay = document.getElementById('ism-e-azam');
            container.innerHTML = "";
            
            if (val === 0) { ismDisplay.innerText = "-"; return; }

            // Find all matches - exact, combinations of 2, and combinations of 3
            let exactMatches = [];
            let pairMatches = [];
            let tripleMatches = [];

            // Find exact matches (single Asma)
            exactMatches = asmaUlHusna.filter(a => a.value === val);

            // Find all combinations of 2
            if (exactMatches.length === 0) {
                const seenPairs = new Set();
                for (let i = 0; i < asmaUlHusna.length; i++) {
                    for (let j = i + 1; j < asmaUlHusna.length; j++) {
                        if (asmaUlHusna[i].value + asmaUlHusna[j].value === val) {
                            const pairKey = `${asmaUlHusna[i].name}|${asmaUlHusna[j].name}`;
                            if (!seenPairs.has(pairKey)) {
                                pairMatches.push([asmaUlHusna[i], asmaUlHusna[j]]);
                                seenPairs.add(pairKey);
                            }
                        }
                    }
                }
            }

            // Find all combinations of 3
            if (exactMatches.length === 0 && pairMatches.length === 0) {
                const seenTriples = new Set();
                for (let i = 0; i < asmaUlHusna.length; i++) {
                    for (let j = i + 1; j < asmaUlHusna.length; j++) {
                        for (let k = j + 1; k < asmaUlHusna.length; k++) {
                            if (asmaUlHusna[i].value + asmaUlHusna[j].value + asmaUlHusna[k].value === val) {
                                const tripleKey = `${asmaUlHusna[i].name}|${asmaUlHusna[j].name}|${asmaUlHusna[k].name}`;
                                if (!seenTriples.has(tripleKey)) {
                                    tripleMatches.push([asmaUlHusna[i], asmaUlHusna[j], asmaUlHusna[k]]);
                                    seenTriples.add(tripleKey);
                                }
                            }
                        }
                    }
                }
            }

            // Function to get unique virtues from matches
            function getUniqueVirtues(matches) {
                const virtuesSet = new Set();
                matches.forEach(match => {
                    if (match.description) {
                        virtuesSet.add(match.description);
                    }
                });
                return Array.from(virtuesSet);
            }

            // Build HTML for results
            let html = '';

            // Show exact matches (single Asma)
            if (exactMatches.length > 0) {
                html += `<div class="combination-section">
                    <div class="combination-title">✨ اسماء حسنیٰ (انفرادی) ✨</div>
                    <div class="combination-grid">`;
                
                exactMatches.forEach(match => {
                    html += `
                        <div class="combination-item">
                            <div class="combination-names">${match.name}</div>
                            <div class="virtues-list">
                                <strong>فضائل:</strong> ${match.description}
                            </div>
                        </div>
                    `;
                });
                
                html += `</div></div>`;
                ismDisplay.innerText = exactMatches.map(m => m.name).join(" / ");
            }
            
            // Show combinations of 2
            else if (pairMatches.length > 0) {
                html += `<div class="combination-section">
                    <div class="combination-title">🔗 دو اسماء کے مجموعے 🔗</div>
                    <div class="combination-grid">`;
                
                pairMatches.forEach(pair => {
                    const uniqueVirtues = getUniqueVirtues(pair);
                    html += `
                        <div class="combination-item">
                            <div class="combination-names">${pair.map(m => m.name).join(" + ")}</div>
                            <div class="virtues-list">
                                <strong>فضائل:</strong><br>
                                ${uniqueVirtues.map(v => `• ${v}`).join("<br>")}
                            </div>
                        </div>
                    `;
                });
                
                html += `</div></div>`;
                ismDisplay.innerText = pairMatches[0].map(m => m.name).join(" + ");
            }
            
            // Show combinations of 3
            else if (tripleMatches.length > 0) {
                html += `<div class="combination-section">
                    <div class="combination-title">🌟 تین اسماء کے مجموعے 🌟</div>
                    <div class="combination-grid">`;
                
                tripleMatches.forEach(triple => {
                    const uniqueVirtues = getUniqueVirtues(triple);
                    html += `
                        <div class="combination-item">
                            <div class="combination-names">${triple.map(m => m.name).join(" + ")}</div>
                            <div class="virtues-list">
                                <strong>فضائل:</strong><br>
                                ${uniqueVirtues.map(v => `• ${v}`).join("<br>")}
                            </div>
                        </div>
                    `;
                });
                
                html += `</div></div>`;
                ismDisplay.innerText = tripleMatches[0].map(m => m.name).join(" + ");
            }
            
            // No matches found
            else {
                html = `<div class="virtue-card" style="text-align: center;">
                    <p style="color: #ff6b6b;">کوئی اسم اعظم نہیں ملا۔ براہ کرم دوسرا عدد یا متن درج کریں۔</p>
                </div>`;
                ismDisplay.innerText = "کوئی اسم نہیں ملا";
            }
            
            container.innerHTML = html;
        }
